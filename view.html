function renderCertificate(data){ function val(v){ return v ? v : "-"; } function relationText(){ if(data.relType && data.relName){ return data.relType + " " + data.relName; } return val(data.relation); } let heirsList = data.heirs || data.heirList || data.legalHeirs || []; let heirsRows = ""; if(heirsList.length > 0){ heirsList.forEach((h,i)=>{ heirsRows += ` ${i+1} ${val(h.name)} ${val(h.age)} ${val(h.rel || h.relation)} ${val(h.addr || h.address)} ${val(h.epic || h.epicNo) || "N/A"} `; }); } else { heirsRows = `No Heir Data`; } document.getElementById("content").innerHTML = `

LEGAL HEIR CERTIFICATE

TO WHOM IT MAY CONCERN

This is to certify that Late ${val(data.deceased)}, ${relationText()}, was a permanent resident of ${val(data.deadAddress)}.

He/She passed away and leaving behind the following legal heirs:


${heirsRows}
Sl Name Age Relation Address EPIC


It is further certified that, upon receipt of the application and after verification of records and field enquiry, it has been found that there are no other legal heirs, successors, or surviving members of the deceased except those mentioned above.

This certificate is issued for official purpose.



Place: Deor Gram Panchayat





___________________________
Signature of Issuing Authority
`; }
